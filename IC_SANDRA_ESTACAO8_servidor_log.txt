bool initOLED() {
  Serial.println("Inicializando OLED...");
  
  Wire.begin(21, 22);
  delay(100);
  
  if(!display.begin(SSD1306_SWITCHCAPVCC, OLED_ADDRESS)) {
    Serial.println("Falha ao inicializar OLED com endereco 0x3C");
    
    if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3D)) {
      Serial.println("Falha ao inicializar OLED com endereco 0x3D");
      Serial.println("Verifique as conexoes:");
      Serial.println("  - VCC -> 3.3V");
      Serial.println("  - GND -> GND");
      Serial.println("  - SDA -> GPIO21");
      Serial.println("  - SCL -> GPIO22");
      return false;
    }
  }
  
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(0, 0);
  display.println("OLED OK!");
  display.display();
  delay(1000);
  
  Serial.println("OLED inicializado com sucesso!");
  return true;
}

void handleLog() {
  String html = "<!DOCTYPE html><html>";
  html += "<head><meta name='viewport' content='width=device-width, initial-scale=1'>";
  html += "<meta http-equiv='refresh' content='120'>";
  html += "<title>Log de Medicoes</title>";
  html += "<style>";
  html += "body { font-family: Arial; margin: 20px; background: #f0f0f0; }";
  html += ".container { max-width: 1000px; margin: auto; background: white; padding: 20px; border-radius: 10px; }";
  html += "h1 { color: #2c3e50; text-align: center; }";
  html += "table { width: 100%; border-collapse: collapse; margin-top: 20px; }";
  html += "th, td { border: 1px solid #ddd; padding: 12px; text-align: left; }";
  html += "th { background-color: #2c3e50; color: white; }";
  html += "tr:nth-child(even) { background-color: #f2f2f2; }";
  html += ".button { background-color: #3498db; color: white; padding: 10px 20px; text-decoration: none; border-radius: 5px; display: inline-block; margin: 10px; }";
  html += "</style>";
  html += "</head><body>";
  html += "<div class='container'>";
  html += "<h1>Historico de Medicoes</h1>";
  
  if (SPIFFS.exists("/logs.txt")) {
    File file = SPIFFS.open("/logs.txt", "r");
    if (file) {
      html += "<table border='1'>";
      html += "<tr><th>Data/Hora</th><th>Temperatura (C)</th><th>Umidade (%)</th></tr>";
      
      String lines[200];
      int count = 0;
      while (file.available() && count < 200) {
        lines[count] = file.readStringUntil('\n');
        lines[count].trim();
        if (lines[count].length() > 0) count++;
      }
      file.close();
      
      for (int i = count - 1; i >= 0; i--) {
        int comma1 = lines[i].indexOf(',');
        int comma2 = lines[i].indexOf(',', comma1 + 1);
        
        if (comma1 > 0 && comma2 > 0) {
          String datetime = lines[i].substring(0, comma1);
          String temp = lines[i].substring(comma1 + 1, comma2);
          String hum = lines[i].substring(comma2 + 1);
          
          html += "<tr>";
          html += "<td>" + datetime + "</td>";
          html += "<td>" + temp + " C</td>";
          html += "<td>" + hum + " %</td>";
          html += "</tr>";
        }
      }
      html += "</table>";
      html += "<p>Total: " + String(count) + " registros</p>";
    }
  } else {
    html += "<p>Nenhum registro encontrado.</p>";
  }
  
  html += "<div style='text-align: center;'>";
  html += "<a href='/' class='button'> Voltar</a>";
  html += "<a href='/download' class='button'> Baixar CSV</a>";
  html += "</div></div></body></html>";
  
  server.send(200, "text/html", html);
}

void handleDownload() {
  if (!SPIFFS.exists("/logs.txt")) {
    server.send(404, "text/plain", "Nenhum log encontrado");
    return;
  }
  
  File file = SPIFFS.open("/logs.txt", "r");
  if (!file) {
    server.send(500, "text/plain", "Erro ao abrir arquivo");
    return;
  }
  
  String content = "Data/Hora,Temperatura (C),Umidade (%)\n";
  while (file.available()) {
    content += file.readStringUntil('\n');
    content += "\n";
  }
  file.close();
  
  server.sendHeader("Content-Type", "text/csv");
  server.sendHeader("Content-Disposition", "attachment; filename=logs_esp32.csv");
  server.send(200, "text/csv", content);
}

void handleClearLog() {
  if (SPIFFS.exists("/logs.txt")) {
    SPIFFS.remove("/logs.txt");
  }
  
  String html = "<!DOCTYPE html><html>";
  html += "<head><meta http-equiv='refresh' content='3;url=/'></head>";
  html += "<body><h2>Log apagado!</h2>";
  html += "<p>Redirecionando...</p></body></html>";
  server.send(200, "text/html", html);
}

void saveLog(float temp, float hum) {
  if (temp == -1 || hum == -1) return;
  
  File file = SPIFFS.open("/logs.txt", FILE_APPEND);
  if (!file) {
    Serial.println("Erro ao abrir arquivo");
    return;
  }
  
  String logEntry = getFormattedDateTime() + "," + String(temp, 1) + "," + String(hum, 1);
  file.println(logEntry);
  file.close();
  
  Serial.println("Log salvo: " + logEntry);
  
  if (oledOK) {
    display.clearDisplay();
    display.drawRect(0, 0, 128, 64, SSD1306_WHITE);
    display.setTextSize(1);
    display.setCursor(25, 15);
    display.println("LOG SALVO!");
    display.setCursor(10, 30);
    display.print("Temp: "); display.print(temp, 1); display.println(" C");
    display.setCursor(10, 45);
    display.print("Umi:  "); display.print(hum, 1); display.println(" %");
    display.display();
    delay(3000);
    updateOLED(temp, hum);
  }
}

String getFormattedDateTime() {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo)) {
    return "00/00/0000 00:00:00";
  }
  
  char buffer[30];
  strftime(buffer, sizeof(buffer), "%d/%m/%Y %H:%M:%S", &timeinfo);
  return String(buffer);
}

void initWiFi() {
  WiFi.mode(WIFI_STA);
  
  Serial.println("\nConfigurando IP fixo...");
  Serial.print("IP desejado: ");
  Serial.println(local_IP);
  Serial.print("Gateway: ");
  Serial.println(gateway);
  Serial.print("Mascara: ");
  Serial.println(subnet);
  
  if (!WiFi.config(local_IP, gateway, subnet, primaryDNS, secondaryDNS)) {
    Serial.println("Falha ao configurar IP fixo! Usando DHCP...");
  } else {
    Serial.println("IP fixo configurado com sucesso!");
  }
  
  WiFi.begin(ssid, password);
  
  if (oledOK) {
    display.clearDisplay();
    display.drawRect(0, 0, 128, 64, SSD1306_WHITE);
    display.setTextSize(1);
    display.setCursor(15, 20);
    display.println("Conectando WiFi...");
    display.setCursor(20, 35);
    display.print("SSID: ");
    display.println(ssid);
    display.setCursor(20, 45);
    display.print("IP Fixo: ");
    display.println(local_IP);
    display.display();
  }
  
  Serial.print("Conectando WiFi");
  
  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 40) {
    delay(500);
    Serial.print(".");
    attempts++;
    
    if (oledOK && attempts % 4 == 0) {
      display.setCursor(15 + (attempts % 20), 55);
      display.print(".");
      display.display();
    }
  }
  
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\nConectado!");
    Serial.print("IP configurado: ");
    Serial.println(WiFi.localIP());
    
    if (WiFi.localIP() == local_IP) {
      Serial.println("IP FIXO ativo: " + WiFi.localIP().toString());
    } else {
      Serial.println("Atencao: IP diferente do configurado!");
      Serial.println("   Configurado: " + local_IP.toString());
      Serial.println("   Obtido: " + WiFi.localIP().toString());
    }
  } else {
    Serial.println("\nFalha WiFi");
    if (oledOK) {
      display.clearDisplay();
      display.drawRect(0, 0, 128, 64, SSD1306_WHITE);
      display.setCursor(20, 25);
      display.println("Falha WiFi!");
      display.setCursor(10, 40);
      display.println("Verifique AP1");
      display.display();
    }
  }
}

void initTime() {
  configTime(gmtOffset_sec, daylightOffset_sec, ntpServer);
  Serial.print("Sincronizando hora");
  
  struct tm timeinfo;
  int attempts = 0;
  while (!getLocalTime(&timeinfo) && attempts < 20) {
    Serial.print(".");
    delay(500);
    attempts++;
  }
  
  if (attempts < 20) {
    Serial.println("\nHora sincronizada!");
    char buffer[30];
    strftime(buffer, sizeof(buffer), "%d/%m/%Y %H:%M:%S", &timeinfo);
    Serial.print("Data: ");
    Serial.println(buffer);
  } else {
    Serial.println("\nFalha na hora");
  }
}

float readDHTTemperature() {
  float t = dht.readTemperature();
  if (isnan(t)) {    
    Serial.println("Falha temperatura!");
    return -1;
  }
  return t;
}

float readDHTHumidity() {
  float h = dht.readHumidity();
  if (isnan(h)) {
    Serial.println("Falha umidade!");
    return -1;
  }
  return h;
}

void setup() {
  Serial.begin(115200);
  delay(1000);
  Serial.println("\n\nIniciando Estacao PAISAGISMO");
  Serial.println("=================================");
  
  oledOK = initOLED();
  
  if (oledOK) {
    showBootScreen();
  } else {
    Serial.println("Sistema rodando sem display OLED");
  }
  
  setupSPIFFS();
  dht.begin();
  initWiFi();
  initTime();
  
  if (oledOK && WiFi.status() == WL_CONNECTED) {
    showIPOnOLED();
  }
  
  server.on("/", handleRoot);
  server.on("/log", handleLog);
  server.on("/download", handleDownload);
  server.on("/clear", handleClearLog);
  server.begin();
  
  Serial.println("\n=================================");
  Serial.println("Servidor HTTP iniciado!");
  Serial.print("Acesse: http://");
  Serial.println(WiFi.localIP());
  Serial.println("=================================\n");
  
  delay(2000);
  
  float temp = readDHTTemperature();
  float hum = readDHTHumidity();
  
  if (temp != -1 && hum != -1) {
    saveLog(temp, hum);
    lastLogTime = millis();
  }
  
  if (oledOK) {
    updateOLED(temp, hum);
  }
  
  Serial.println("Sistema pronto!");
}

void loop() {
  server.handleClient();
  
  if (millis() - lastLogTime >= logInterval) {
    float temp = readDHTTemperature();
    float hum = readDHTHumidity();
    
    if (temp != -1 && hum != -1) {
      saveLog(temp, hum);
      if (oledOK) updateOLED(temp, hum);
      lastLogTime = millis();
      Serial.println("Log automatico (2h)");
    } else {
      lastLogTime = millis() - (logInterval - 300000);
    }
  }
  
  static unsigned long lastOLEDUpdate = 0;
  if (oledOK && (millis() - lastOLEDUpdate >= 10000)) {
    float temp = readDHTTemperature();
    float hum = readDHTHumidity();
    if (temp != -1 && hum != -1) {
      updateOLED(temp, hum);
    }
    lastOLEDUpdate = millis();
  }
  
  delay(100);
}
