bool initOLED() {
  Serial.println("Inicializando OLED...");
  Wire.begin(21, 22);
  delay(100);
  
  if(!display.begin(SSD1306_SWITCHCAPVCC, OLED_ADDRESS)) {
    if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3D)) {
      Serial.println("OLED nao encontrado - rodando sem display");
      return false;
    }
  }
  
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(0, 0);
  display.println("OLED OK!");
  display.display();
  delay(1000);
  return true;
}

void handleLog() {
  String html = "<!DOCTYPE html><html><head><meta name='viewport' content='width=device-width, initial-scale=1'>";
  html += "<title>Log de Medicoes</title><style>";
  html += "body { font-family: Arial; margin: 20px; background: #f0f0f0; }";
  html += ".container { max-width: 1000px; margin: auto; background: white; padding: 20px; border-radius: 10px; }";
  html += "h1 { color: #2c3e50; text-align: center; }";
  html += "table { width: 100%; border-collapse: collapse; margin-top: 20px; }";
  html += "th, td { border: 1px solid #ddd; padding: 12px; text-align: left; }";
  html += "th { background-color: #2c3e50; color: white; }";
  html += ".button { background-color: #3498db; color: white; padding: 10px 20px; text-decoration: none; border-radius: 5px; display: inline-block; margin: 10px; }";
  html += "</style></head><body><div class='container'>";
  html += "<h1>Historico de Medicoes</h1>";
  
  if (SPIFFS.exists("/logs.txt")) {
    File file = SPIFFS.open("/logs.txt", "r");
    if (file) {
      html += "<table>";
      html += "<tr><th>Data/Hora</th><th>Temperatura</th><th>Umidade</th></tr>";
      
      String lines[200];
      int count = 0;
      while (file.available() && count < 200) {
        lines[count] = file.readStringUntil('\n');
        lines[count].trim();
        if (lines[count].length() > 0) count++;
      }
      file.close();
      
      for (int i = count - 1; i >= 0; i--) {
        int comma1 = lines[i].indexOf(',');
        int comma2 = lines[i].indexOf(',', comma1 + 1);
        if (comma1 > 0 && comma2 > 0) {
          html += "<tr>";
          html += "<td>" + lines[i].substring(0, comma1) + "</td>";
          html += "<td>" + lines[i].substring(comma1 + 1, comma2) + " °C</td>";
          html += "<td>" + lines[i].substring(comma2 + 1) + " %</td>";
          html += "</tr>";
        }
      }
      html += "</table><p>Total: " + String(count) + " registros</p>";
    }
  } else {
    html += "<p>Nenhum registro encontrado.</p>";
  }
  
  html += "<div style='text-align: center;'><a href='/' class='button'> Voltar</a>";
  html += "<a href='/download' class='button'> Baixar CSV</a></div></div></body></html>";
  server.send(200, "text/html", html);
}

void handleDownload() {
  if (!SPIFFS.exists("/logs.txt")) {
    server.send(404, "text/plain", "Nenhum log encontrado");
    return;
  }
  
  File file = SPIFFS.open("/logs.txt", "r");
  if (!file) {
    server.send(500, "text/plain", "Erro ao abrir arquivo");
    return;
  }
  
  String content = "Data/Hora,Temperatura (C),Umidade (%)\n";
  while (file.available()) {
    content += file.readStringUntil('\n');
    content += "\n";
  }
  file.close();
  
  server.sendHeader("Content-Type", "text/csv");
  server.sendHeader("Content-Disposition", "attachment; filename=logs_esp32.csv");
  server.send(200, "text/csv", content);
}

  html += "<body><h2>Log apagado!</h2><p>Redirecionando...</p></body></html>";
  server.send(200, "text/html", html);
}

void saveLog(float temp, float hum) {
  if (temp == -1 || hum == -1) return;
  
  File file = SPIFFS.open("/logs.txt", FILE_APPEND);
  if (!file) {
    Serial.println("Erro ao abrir arquivo de log");
    return;
  }
  
  String logEntry = getFormattedDateTime() + "," + String(temp, 1) + "," + String(hum, 1);
  file.println(logEntry);
  file.close();
  Serial.println("Log salvo: " + logEntry);
}

String getFormattedDateTime() {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo)) {
    return "00/00/0000 00:00:00";
  }
  char buffer[30];
  strftime(buffer, sizeof(buffer), "%d/%m/%Y %H:%M:%S", &timeinfo);
  return String(buffer);
}

bool loadConfig() {
  if (!SPIFFS.exists("/config.json")) {
    Serial.println("Arquivo de configuracao nao encontrado!");
    hasWifiConfig = false;
    return false;
  }
  
  File file = SPIFFS.open("/config.json", "r");
  if (!file) {
    Serial.println("Erro ao abrir config.json");
    return false;
  }
  
  StaticJsonDocument<512> doc;
  DeserializationError error = deserializeJson(doc, file);
  file.close();
  
  if (error) {
    Serial.println("Erro ao ler config.json");
    return false;
  }
  
  ssid = doc["ssid"].as<String>();
  password = doc["password"].as<String>();
  useStaticIP = doc["useStaticIP"] | false;
  
  if (useStaticIP) {
    local_IP.fromString(doc["ip"].as<String>());
    gateway.fromString(doc["gateway"].as<String>());
    subnet.fromString(doc["subnet"].as<String>());
    primaryDNS.fromString(doc["dns1"].as<String>());
    secondaryDNS.fromString(doc["dns2"].as<String>());
  }
  
  hasWifiConfig = (ssid.length() > 0);
  Serial.println("Configuracoes carregadas:");
  Serial.println("  SSID: " + ssid);
  Serial.println("  IP: " + String(useStaticIP ? "Estatico" : "DHCP"));
  
  return true;
}

bool saveConfig() {
  StaticJsonDocument<512> doc;
  doc["ssid"] = ssid;
  doc["password"] = password;
  doc["useStaticIP"] = useStaticIP;
  
  if (useStaticIP) {
    doc["ip"] = local_IP.toString();
    doc["gateway"] = gateway.toString();
    doc["subnet"] = subnet.toString();
    doc["dns1"] = primaryDNS.toString();
    doc["dns2"] = secondaryDNS.toString();
  }
  
  File file = SPIFFS.open("/config.json", "w");
  if (!file) {
    Serial.println("Erro ao salvar config.json");
    return false;
  }
  
  if (serializeJson(doc, file) == 0) {
    Serial.println("Erro ao escrever JSON");
    file.close();
    return false;
  }
  
  file.close();
  Serial.println("Configuracoes salvas!");
  return true;
}

bool initWiFi() {
  if (!hasWifiConfig || ssid.length() == 0) {
    Serial.println("\n⚠️  Nenhuma configuração WiFi encontrada!");
    Serial.println("Digite 'CONFIG' para configurar a rede");
    return false;
  }
  
  WiFi.mode(WIFI_STA);
  
  if (useStaticIP) {
    Serial.println("\nConfigurando IP Estatico:");
    Serial.println("  IP: " + local_IP.toString());
    Serial.println("  Gateway: " + gateway.toString());
    Serial.println("  Mascara: " + subnet.toString());
    
    if (!WiFi.config(local_IP, gateway, subnet, primaryDNS, secondaryDNS)) {
      Serial.println("  ⚠️  Falha ao configurar IP estatico! Usando DHCP...");
      useStaticIP = false;
    } else {
      Serial.println("  ✅ IP estatico configurado");
    }
  } else {
    Serial.println("\nUsando DHCP - IP automatico");
  }
  
  Serial.print("Conectando ao WiFi '" + ssid + "'");
  WiFi.begin(ssid.c_str(), password.c_str());
  
  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 30) {
    delay(500);
    Serial.print(".");
    attempts++;
    
    if (oledOK && attempts % 4 == 0) {
      display.clearDisplay();
      display.setCursor(10, 30);
      display.print("Conectando...");
      display.display();
    }
  }
  
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n✅ Conectado!");
    Serial.println("  IP: " + WiFi.localIP().toString());
    Serial.println("  MAC: " + WiFi.macAddress());
    return true;
  } else {
    Serial.println("\n❌ Falha na conexao!");
    Serial.println("  Verifique SSID e senha");
    return false;
  }
}

void initTime() {
  configTime(gmtOffset_sec, daylightOffset_sec, ntpServer);
  Serial.print("Sincronizando hora");
  
  struct tm timeinfo;
  int attempts = 0;
  while (!getLocalTime(&timeinfo) && attempts < 20) {
    Serial.print(".");
    delay(500);
    attempts++;
  }
  
  if (attempts < 20) {
    Serial.println("\n✅ Hora sincronizada!");
    char buffer[30];
    strftime(buffer, sizeof(buffer), "%d/%m/%Y %H:%M:%S", &timeinfo);
    Serial.println("  Data: " + String(buffer));
  } else {
    Serial.println("\n⚠️  Falha na sincronizacao - hora nao confiavel");
  }
}

float readDHTTemperature() {
  float t = dht.readTemperature();
  if (isnan(t)) {
    Serial.println("Erro na leitura da temperatura");
    return -1;
  }
  return t;
}

float readDHTHumidity() {
  float h = dht.readHumidity();
  if (isnan(h)) {
    Serial.println("Erro na leitura da umidade");
    return -1;
  }
  return h;
}

void configureWiFi() {
  Serial.println("\n--- CONFIGURACAO WiFi ---");
  Serial.print("Digite o SSID da rede: ");
  
  String newSSID = "";
  while (newSSID.length() == 0) {
    while (!Serial.available()) {
      delay(100);
    }
    newSSID = Serial.readStringUntil('\n');
    newSSID.trim();
  }
  
  Serial.print("Digite a senha: ");
  String newPassword = "";
  while (newPassword.length() == 0) {
    while (!Serial.available()) {
      delay(100);
    }
    newPassword = Serial.readStringUntil('\n');
    newPassword.trim();
  }
  
  ssid = newSSID;
  password = newPassword;
  hasWifiConfig = true;
  useStaticIP = false;  // Começa com DHCP
  
  if (saveConfig()) {
    Serial.println("\n✅ Configuracoes salvas!");
    Serial.println("   Reiniciando para aplicar...");
    delay(2000);
    ESP.restart();
  } else {
    Serial.println("\n❌ Erro ao salvar configuracoes!");
  }
}

void configureStaticIP() {
  Serial.println("\n--- CONFIGURACAO IP ESTATICO ---");
  Serial.println("Digite as configuracoes:");
  
  Serial.print("IP (ex: 192.168.0.100): ");
  String ipStr = "";
  while (ipStr.length() == 0) {
    while (!Serial.available()) delay(100);
    ipStr = Serial.readStringUntil('\n');
    ipStr.trim();
  }
  
  Serial.print("Gateway (ex: 192.168.0.1): ");
  String gwStr = "";
  while (gwStr.length() == 0) {
    while (!Serial.available()) delay(100);
    gwStr = Serial.readStringUntil('\n');
    gwStr.trim();
  }
  
  Serial.print("Mascara (ex: 255.255.255.0): ");
  String maskStr = "";
  while (maskStr.length() == 0) {
    while (!Serial.available()) delay(100);
    maskStr = Serial.readStringUntil('\n');
    maskStr.trim();
  }
  
  Serial.print("DNS Primario (ex: 8.8.8.8): ");
  String dns1Str = "";
  while (dns1Str.length() == 0) {
    while (!Serial.available()) delay(100);
    dns1Str = Serial.readStringUntil('\n');
    dns1Str.trim();
  }
  
  Serial.print("DNS Secundario (ex: 8.8.4.4): ");
  String dns2Str = "";
  while (dns2Str.length() == 0) {
    while (!Serial.available()) delay(100);
    dns2Str = Serial.readStringUntil('\n');
    dns2Str.trim();
  }
  
  if (local_IP.fromString(ipStr) && gateway.fromString(gwStr) && 
      subnet.fromString(maskStr) && primaryDNS.fromString(dns1Str) && 
      secondaryDNS.fromString(dns2Str)) {
    
    useStaticIP = true;
    if (saveConfig()) {
      Serial.println("\n✅ Configuracoes IP salvas!");
      Serial.println("   Reiniciando para aplicar...");
      delay(2000);
      ESP.restart();
    } else {
      Serial.println("\n❌ Erro ao salvar!");
    }
  } else {
    Serial.println("\n❌ IP invalido! Use o formato correto (ex: 192.168.0.100)");
  }
}

void processSerialCommands() {
  if (Serial.available()) {
    String input = Serial.readStringUntil('\n');
    input.trim();
    input.toUpperCase();
    
    if (input == "CONFIG" || input == "MENU") {
      configMode = true;
      showConfigMenu();
      return;
    }
    
    if (configMode) {
      int option = input.toInt();
      switch(option) {
        case 1:
          configMode = false;
          configureWiFi();
          break;
        case 2:
          configMode = false;
          configureStaticIP();
          break;
        case 3:
          configMode = false;
          useDHCP();
          break;
        case 4:
          showCurrentConfig();
          showConfigMenu();
          break;
        case 5:
          testWiFi();
          showConfigMenu();
          break;
        case 6:
          Serial.println("Reiniciando ESP32...");
          delay(1000);
          ESP.restart();
          break;
        case 0:
          configMode = false;
          Serial.println("Saindo do menu...");
          break;
        default:
          Serial.println("Opcao invalida!");
          showConfigMenu();
      }
    }
  }
}

void setup() {
  Serial.begin(115200);
  delay(1000);

  Serial.println("║        FATEC SP                       ║");
  Serial.println("╚════════════════════════════════════════╝");
  
  setupSPIFFS();
  loadConfig();
  
  oledOK = initOLED();
  if (oledOK) showBootScreen();
  
  dht.begin();
  
  // Tenta conectar ao WiFi
  if (initWiFi()) {
    initTime();
    if (oledOK) showIPOnOLED();
    
    // Inicia servidor web
    server.on("/", handleRoot);
    server.on("/log", handleLog);
    server.on("/download", handleDownload);
    server.on("/clear", handleClearLog);
    server.begin();
    
    Serial.println("\n╔════════════════════════════════════════╗");
    Serial.println("║   SERVIDOR HTTP INICIADO!             ║");
    Serial.println("║   Acesse: http://" + WiFi.localIP().toString());
    Serial.println("╚════════════════════════════════════════╝");
    
    // Primeira leitura
    float temp = readDHTTemperature();
    float hum = readDHTHumidity();
    if (temp != -1 && hum != -1) {
      saveLog(temp, hum);
      lastLogTime = millis();
      if (oledOK) updateOLED(temp, hum);
    }
  }
  
  Serial.println("\n💡 Digite 'CONFIG' para acessar o menu de configuracoes");
}

void loop() {
  if (WiFi.status() == WL_CONNECTED) {
    server.handleClient();
  }
  
  // Log automatico a cada 2 horas
  if (WiFi.status() == WL_CONNECTED && millis() - lastLogTime >= logInterval) {
    float temp = readDHTTemperature();
    float hum = readDHTHumidity();
    if (temp != -1 && hum != -1) {
      saveLog(temp, hum);
      if (oledOK) updateOLED(temp, hum);
      lastLogTime = millis();
      Serial.println("Log automatico (2h)");
    }
  }
  
  // Atualiza display a cada 10 segundos
  static unsigned long lastOLEDUpdate = 0;
  if (oledOK && WiFi.status() == WL_CONNECTED && (millis() - lastOLEDUpdate >= 10000)) {
    float temp = readDHTTemperature();
    float hum = readDHTHumidity();
    if (temp != -1 && hum != -1) {
      updateOLED(temp, hum);
    }
    lastOLEDUpdate = millis();
  }
  
  // Processa comandos do serial
  processSerialCommands();
  
  delay(100);
}
